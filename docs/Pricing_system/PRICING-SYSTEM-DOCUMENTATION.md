# 1. EXECUTIVE SUMMARY

## 1.1 Purpose
A comprehensive pricing calculator that determines accurate hourly rates for a service business by accounting for:
- True labor costs (burden, benefits, taxes)
- Unproductive time / efficiency losses
- Vehicle/fleet operating costs
- Operating overhead expenses
- Target profit margins

## 1.2 Core Problem Solved
**Before:** Guessing at hourly rates, not accounting for true labor burden, underpricing services
**After:** Data-driven pricing that guarantees profitability at every job type

## 1.3 Key Outputs
| Output | Description |
|--------|-------------|
| True Cost Per Billable Hour | What each tech actually costs per productive hour |
| Loaded Cost Per Hour | Labor + Vehicle cost combined |
| Sell Rate Per Hour | Rate needed to hit target margin |
| Break-Even Point | Monthly revenue needed to cover all costs |
| Net Profit Projection | Expected annual profit at target revenue |

## 7.1 Core Calculation Functions

```javascript
// ============================================================================
// FORMATTING UTILITIES
// ============================================================================

export const formatCurrency = (num) => 
  new Intl.NumberFormat('en-US', { 
    style: 'currency', 
    currency: 'USD', 
    minimumFractionDigits: 2 
  }).format(num || 0);

export const formatCurrencyShort = (num) => {
  if (Math.abs(num) >= 1000000) return `$${(num / 1000000).toFixed(1)}M`;
  if (Math.abs(num) >= 1000) return `$${(num / 1000).toFixed(1)}K`;
  return formatCurrency(num);
};

export const formatPercent = (num) => `${(num || 0).toFixed(2)}%`;

export const formatNumber = (num) => 
  new Intl.NumberFormat('en-US').format(num || 0);

// ============================================================================
// BURDEN CALCULATIONS
// ============================================================================

/**
 * Calculate total annual burden for an employee
 * @param {number} annualBasePay - Annual base pay
 * @param {object} rates - Burden rate percentages
 * @param {object} benefits - Monthly benefit amounts
 * @returns {object} Burden breakdown
 */
export const calculateBurden = (annualBasePay, rates, benefits) => {
  const payrollTaxes = annualBasePay * (rates.payrollTaxRate / 100);
  const futa = Math.min(annualBasePay, 7000) * (rates.futaRate / 100);
  const suta = annualBasePay * (rates.sutaRate / 100);
  const workersComp = annualBasePay * (rates.workersCompRate / 100);
  const healthInsuranceAnnual = benefits.healthInsurance * 12;
  const dentalVisionAnnual = benefits.dentalVision * 12;
  const retirement401kAnnual = annualBasePay * (rates.retirement401k / 100);
  const otherBenefitsAnnual = benefits.otherBenefits * 12;
  
  const totalBurdenAnnual = 
    payrollTaxes + futa + suta + workersComp + 
    healthInsuranceAnnual + dentalVisionAnnual + 
    retirement401kAnnual + otherBenefitsAnnual;
  
  const burdenPercent = (totalBurdenAnnual / annualBasePay) * 100;
  
  return {
    payrollTaxes,
    futa,
    suta,
    workersComp,
    healthInsuranceAnnual,
    dentalVisionAnnual,
    retirement401kAnnual,
    otherBenefitsAnnual,
    totalBurdenAnnual,
    burdenPercent
  };
};

// ============================================================================
// EFFICIENCY CALCULATIONS
// ============================================================================

/**
 * Calculate productivity metrics from unproductive time
 * @param {number} paidHoursPerDay - Total paid hours per day
 * @param {array} unproductiveTime - Array of time entries
 * @param {number} workingDaysPerYear - Working days per year
 * @returns {object} Productivity metrics
 */
export const calculateProductivity = (paidHoursPerDay, unproductiveTime, workingDaysPerYear) => {
  const paidUnproductiveHours = unproductiveTime
    .filter(t => t.paid)
    .reduce((sum, t) => sum + t.hours, 0);
    
  const unpaidUnproductiveHours = unproductiveTime
    .filter(t => !t.paid)
    .reduce((sum, t) => sum + t.hours, 0);
    
  const totalUnproductiveHours = paidUnproductiveHours + unpaidUnproductiveHours;
  const billableHoursPerDay = paidHoursPerDay - paidUnproductiveHours;
  const billableHoursPerYear = billableHoursPerDay * workingDaysPerYear;
  const efficiencyPercent = (billableHoursPerDay / paidHoursPerDay) * 100;
  
  return {
    paidUnproductiveHours,
    unpaidUnproductiveHours,
    totalUnproductiveHours,
    billableHoursPerDay,
    billableHoursPerYear,
    efficiencyPercent
  };
};

// ============================================================================
// TRUE COST CALCULATION
// ============================================================================

/**
 * Calculate true cost per billable hour
 * @param {number} totalCostAnnual - Total annual cost (base + burden)
 * @param {number} billableHoursPerYear - Billable hours per year
 * @returns {number} True cost per billable hour
 */
export const calculateTrueCostPerHour = (totalCostAnnual, billableHoursPerYear) => {
  if (billableHoursPerYear <= 0) return 0;
  return totalCostAnnual / billableHoursPerYear;
};

// ============================================================================
// VEHICLE COST CALCULATION
// ============================================================================

/**
 * Calculate vehicle cost per billable hour
 * @param {number} totalMonthlyFleetCost - Total monthly fleet cost
 * @param {number} totalBillableHoursPerMonth - Total billable hours per month (all techs)
 * @returns {number} Vehicle cost per billable hour
 */
export const calculateVehicleCostPerHour = (totalMonthlyFleetCost, totalBillableHoursPerMonth) => {
  if (totalBillableHoursPerMonth <= 0) return 0;
  return totalMonthlyFleetCost / totalBillableHoursPerMonth;
};

// ============================================================================
// HOURLY RATE CALCULATION
// ============================================================================

/**
 * Calculate sell rate from cost and target margin
 * @param {number} loadedCostPerHour - Total cost per billable hour (labor + vehicle)
 * @param {number} targetMarginPercent - Target gross margin percentage
 * @returns {number} Hourly sell rate
 */
export const calculateHourlyRate = (loadedCostPerHour, targetMarginPercent) => {
  if (targetMarginPercent >= 100) return Infinity;
  return loadedCostPerHour / (1 - targetMarginPercent / 100);
};

/**
 * Calculate member rate with discount
 * @param {number} hourlyRate - Standard hourly rate
 * @param {number} discountPercent - Member discount percentage
 * @returns {number} Discounted member rate
 */
export const calculateMemberRate = (hourlyRate, discountPercent) => {
  return hourlyRate * (1 - discountPercent / 100);
};

/**
 * Calculate minimum invoice amount
 * @param {number} hourlyRate - Hourly rate
 * @param {number} minHours - Minimum hours
 * @param {number} surcharge - Flat surcharge
 * @returns {number} Minimum invoice amount
 */
export const calculateMinInvoice = (hourlyRate, minHours, surcharge) => {
  return (hourlyRate * minHours) + surcharge;
};

// ============================================================================
// MARKUP CALCULATIONS
// ============================================================================

/**
 * Calculate markup percentage from gross margin
 * @param {number} marginPercent - Gross margin percentage
 * @returns {number} Markup percentage
 */
export const calcMarkupFromMargin = (marginPercent) => {
  if (marginPercent >= 100) return Infinity;
  return marginPercent / (100 - marginPercent) * 100;
};

/**
 * Calculate gross margin from markup percentage
 * @param {number} markupPercent - Markup percentage
 * @returns {number} Gross margin percentage
 */
export const calcMarginFromMarkup = (markupPercent) => {
  return markupPercent / (100 + markupPercent) * 100;
};

/**
 * Calculate multiplier from gross margin
 * @param {number} marginPercent - Gross margin percentage
 * @returns {number} Price multiplier
 */
export const calcMultiplierFromMargin = (marginPercent) => {
  if (marginPercent >= 100) return Infinity;
  return 1 / (1 - marginPercent / 100);
};

/**
 * Calculate sell price from cost and margin
 * @param {number} cost - Item cost
 * @param {number} marginPercent - Target gross margin
 * @returns {number} Sell price
 */
export const calcSellPrice = (cost, marginPercent) => {
  if (marginPercent >= 100) return Infinity;
  return cost / (1 - marginPercent / 100);
};

/**
 * Find applicable markup tier for a cost
 * @param {number} cost - Item cost
 * @param {array} tiers - Array of markup tiers
 * @returns {object} Applicable tier
 */
export const findMarkupTier = (cost, tiers) => {
  return tiers.find(t => cost >= t.minCost && cost <= t.maxCost) 
    || tiers[tiers.length - 1];
};

// ============================================================================
// P&L CALCULATIONS
// ============================================================================

/**
 * Calculate P&L projections
 * @param {object} params - Calculation parameters
 * @returns {object} P&L breakdown
 */
export const calculatePL = ({
  yearlyRevenue,
  materialCostPercent,
  laborCostAnnual,
  overheadAnnual,
  techCount
}) => {
  const materialsCost = yearlyRevenue * (materialCostPercent / 100);
  const grossProfit = yearlyRevenue - materialsCost - laborCostAnnual;
  const grossProfitPercent = yearlyRevenue > 0 ? (grossProfit / yearlyRevenue) * 100 : 0;
  const netProfit = grossProfit - overheadAnnual;
  const netProfitPercent = yearlyRevenue > 0 ? (netProfit / yearlyRevenue) * 100 : 0;
  const laborPercent = yearlyRevenue > 0 ? (laborCostAnnual / yearlyRevenue) * 100 : 0;
  const overheadPercent = yearlyRevenue > 0 ? (overheadAnnual / yearlyRevenue) * 100 : 0;
  
  return {
    yearlyRevenue,
    monthlyRevenue: yearlyRevenue / 12,
    weeklyRevenue: yearlyRevenue / 52,
    materialsCost,
    materialsPercent: materialCostPercent,
    laborCost: laborCostAnnual,
    laborPercent,
    grossProfit,
    grossProfitPercent,
    overhead: overheadAnnual,
    overheadPercent,
    netProfit,
    netProfitPercent,
    revenuePerTech: techCount > 0 ? yearlyRevenue / techCount : 0
  };
};

/**
 * Calculate break-even point
 * @param {number} monthlyOverhead - Monthly overhead costs
 * @param {number} monthlyLaborCost - Monthly labor costs
 * @param {number} materialCostPercent - Materials as % of revenue
 * @returns {object} Break-even metrics
 */
export const calculateBreakEven = (monthlyOverhead, monthlyLaborCost, materialCostPercent) => {
  const contributionMargin = 1 - (materialCostPercent / 100);
  const monthlyBreakEven = (monthlyOverhead + monthlyLaborCost) / contributionMargin;
  
  return {
    monthlyBreakEven,
    dailyBreakEven: monthlyBreakEven / 21.67, // Average working days per month
    weeklyBreakEven: monthlyBreakEven / 4.33
  };
};
```

```javascript
// ============================================================================
// useTechnicianCalculations.js

import { useMemo } from 'react';
import { calculateBurden, calculateProductivity, calculateTrueCostPerHour } from '../utils/calculations';

export const useTechnicianCalculations = (technicians, vehicles, settings) => {
  return useMemo(() => {
    return technicians.map(tech => {
      const hoursPerYear = tech.paidHoursPerDay * settings.workingDaysPerYear;
      const annualBasePay = tech.basePayRate * hoursPerYear;
      
      // Calculate burden
      const burden = calculateBurden(
        annualBasePay,
        {
          payrollTaxRate: tech.payrollTaxRate,
          futaRate: tech.futaRate,
          sutaRate: tech.sutaRate,
          workersCompRate: tech.workersCompRate,
          retirement401k: tech.retirement401k
        },
        {
          healthInsurance: tech.healthInsurance,
          dentalVision: tech.dentalVision,
          otherBenefits: tech.otherBenefits
        }
      );
      
      const totalCostAnnual = annualBasePay + burden.totalBurdenAnnual;
      const burdenedHourlyRate = totalCostAnnual / hoursPerYear;
      
      // Calculate productivity
      const productivity = calculateProductivity(
        tech.paidHoursPerDay,
        tech.unproductiveTime,
        settings.workingDaysPerYear
      );
      
      // True cost per billable hour
      const trueCostPerBillableHour = calculateTrueCostPerHour(
        totalCostAnnual,
        productivity.billableHoursPerYear
      );
      
      // Vehicle costs
      const assignedVehicle = vehicles.find(v => v.id === tech.assignedVehicleId);
      const vehicleCostMonthly = assignedVehicle
        ? (assignedVehicle.monthlyPayment || 0) + 
          (assignedVehicle.insurance || 0) + 
          (assignedVehicle.fuel || 0) + 
          (assignedVehicle.maintenance || 0)
        : 0;
      const vehicleCostPerBillableHour = vehicleCostMonthly / 
        (productivity.billableHoursPerDay * (settings.workingDaysPerYear / 12));
      
      // Total loaded cost
      const totalLoadedCostPerHour = trueCostPerBillableHour + vehicleCostPerBillableHour;
      
      return {
        ...tech,
        hoursPerYear,
        hoursPerWeek: tech.paidHoursPerDay * 5,
        annualBasePay,
        ...burden,
        totalCostAnnual,
        burdenedHourlyRate,
        ...productivity,
        trueCostPerBillableHour,
        assignedVehicle,
        vehicleCostMonthly,
        vehicleCostPerBillableHour,
        totalLoadedCostPerHour,
        monthlyCost: totalCostAnnual / 12
      };
    });
  }, [technicians, vehicles, settings]);
};
```

# APPENDIX B: GLOSSARY

| Term | Definition |
|------|------------|
| **Base Pay Rate** | Hourly wage before any burden or benefits |
| **Burden** | Employer costs beyond base pay (taxes, insurance, benefits) |
| **Burden Percent** | Total burden divided by base pay, expressed as percentage |
| **Burdened Rate** | Base pay + burden, per hour |
| **Billable Hours** | Hours that can be charged to customers |
| **Efficiency** | Billable hours divided by paid hours |
| **True Cost Per Billable Hour** | Total annual cost divided by billable hours per year |
| **Loaded Cost** | True cost plus vehicle/equipment costs |
| **Sell Rate** | Price charged to customers per hour |
| **Gross Margin** | (Revenue - Cost) / Revenue as percentage |
| **Markup** | (Sell - Cost) / Cost as percentage |
| **Overhead** | Operating costs not directly tied to jobs |
| **Break-Even** | Revenue needed to cover all costs (zero profit) |
